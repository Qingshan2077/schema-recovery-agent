    """
    鐩戞帶璁板綍鍣?鈥?灏嗘瘡娆″垎鏋愮殑鎬ц兘鏁版嵁鍜岀粨鏋滄寔涔呭寲銆?    
    璁板綍鐨勫唴瀹癸細
    - 姣忔鍒嗘瀽鐨勫畬鏁存楠ゆ棩蹇?    - 姣忎釜 Worker 鐨勮€楁椂銆佸伐鍏疯皟鐢ㄦ鏁般€佹垚鍔?澶辫触
    - 鏈€缁堣緭鍑哄叧绯绘暟閲忓拰缃俊搴﹀垎甯?    - 璇佹嵁婧愯础鐚巼
    """

    def __init__(self, db_path: str = None):
        if db_path is None:
            db_path = os.path.join(
                os.environ.get("DATA_DIR", "/app/data"),
                "monitor.db"
            )
        self.db_path = db_path
        self._init_db()

    def _init_db(self):
        os.makedirs(os.path.dirname(self.db_path), exist_ok=True)
        conn = sqlite3.connect(self.db_path)
        try:
            conn.execute("""
                CREATE TABLE IF NOT EXISTS analysis_records (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    session_id TEXT NOT NULL UNIQUE,
                    status TEXT NOT NULL,          -- completed / error
                    total_duration_ms INTEGER,
                    total_tool_calls INTEGER,
                    table_count INTEGER,
                    high_confidence_count INTEGER,
                    medium_confidence_count INTEGER,
                    low_confidence_count INTEGER,
                    evidence_contributions TEXT,    -- JSON
                    total_steps INTEGER,
                    created_at TEXT DEFAULT (datetime('now'))
                )
            """)
            conn.execute("""
                CREATE TABLE IF NOT EXISTS worker_records (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    session_id TEXT NOT NULL,
                    worker_id TEXT NOT NULL,
                    status TEXT NOT NULL,           -- success / error / skipped / partial
                    duration_ms INTEGER,
                    tool_call_count INTEGER,
                    error TEXT,
                    created_at TEXT DEFAULT (datetime('now'))
                )
            """)
            conn.commit()
        finally:
            conn.close()

    def record_analysis(self, session_id: str, context: dict, steps: list[dict]):
        """璁板綍涓€娆″垎鏋愮殑瀹屾暣鏁版嵁"""
        conn = sqlite3.connect(self.db_path)
        try:
            # 姹囨€讳俊鎭?            merge_result = context.get("merge_result", {})
            summary = merge_result.get("summary", {})
            survey_result = context.get("survey_result", {})
            survey_summary = survey_result.get("summary", {})

            total_duration = sum(
                s.get("duration_ms", 0) for s in steps
                if s.get("worker") != "router"
            )
            total_tool_calls = sum(
                len(s.get("tool_calls", [])) for s in steps
                if "tool_calls" in s
            )
            contributions = merge_result.get("source_contributions", {})

            # 鍐欏叆鍒嗘瀽璁板綍
            conn.execute("""
                INSERT OR REPLACE INTO analysis_records
                    (session_id, status, total_duration_ms, total_tool_calls,
                     table_count, high_confidence_count, medium_confidence_count,
                     low_confidence_count, evidence_contributions, total_steps)
                VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
            """, (
                session_id,
                "completed",
                total_duration,
                total_tool_calls,
                survey_summary.get("total_tables", 0),
                summary.get("high_confidence", 0),
                summary.get("medium_confidence", 0),
                summary.get("low_confidence", 0),
                json.dumps(contributions, ensure_ascii=False),
                len(steps),
            ))

            # 閫?Worker 璁板綍
            for step in steps:
                if step.get("worker") == "router" or step.get("worker") == "memory":
                    continue
                conn.execute("""
                    INSERT INTO worker_records
                        (session_id, worker_id, status, duration_ms,
                         tool_call_count, error)
                    VALUES (?, ?, ?, ?, ?, ?)
                """, (
                    session_id,
                    step.get("worker", "unknown"),
                    step.get("status", "unknown"),
                    step.get("duration_ms", 0),
                    len(step.get("tool_calls", [])),
                    step.get("error"),
                ))

            conn.commit()
        finally:
            conn.close()

    def get_stats(self) -> dict:
        """鑾峰彇鐩戞帶缁熻鎽樿"""
        conn = sqlite3.connect(self.db_path)
        try:
            total = conn.execute("SELECT COUNT(*) FROM analysis_records").fetchone()[0]
            if total == 0:
                return {"total_analyses": 0, "message": "No analysis records yet"}

            avg_duration = conn.execute(
                "SELECT AVG(total_duration_ms) FROM analysis_records"
            ).fetchone()[0] or 0
            avg_tables = conn.execute(
                "SELECT AVG(table_count) FROM analysis_records"
            ).fetchone()[0] or 0

            worker_avg = conn.execute("""
                SELECT worker_id,
                       COUNT(*) as runs,
                       ROUND(AVG(duration_ms)) as avg_duration,
                       ROUND(100.0 * SUM(CASE WHEN status = 'success' THEN 1 ELSE 0 END) / COUNT(*), 1) as success_rate
                FROM worker_records
                GROUP BY worker_id
            """).fetchall()

            return {
                "total_analyses": total,
                "avg_duration_ms": round(avg_duration, 0),
                "avg_tables_per_analysis": round(avg_tables, 1),
                "worker_stats": [
                    {
                        "worker_id": w[0],
                        "runs": w[1],
                        "avg_duration_ms": w[2],
                        "success_rate": w[3],
                    }
                    for w in worker_avg
                ],
                "recent_analyses": [
                    {
                        "session_id": row[1],
                        "status": row[2],
                        "duration_ms": row[3],
                        "high_confidence": row[5],
                        "date": row[10],
                    }
                    for row in conn.execute("""
                        SELECT * FROM analysis_records ORDER BY created_at DESC LIMIT 5
                    """).fetchall()
                ],
            }
        finally:
            conn.close()

    def __init__(self, db_path: str = None):
        if db_path is None:

    def get_evidence_source_overview(self) -> dict:
        """杩斿洖鍚勮瘉鎹簮鐨勮础鐚秼鍔?""
        conn = sqlite3.connect(self.db_path)
        try:
            # 浠庢渶杩?10 娆″垎鏋愪腑鎻愬彇璇佹嵁璐＄尞鐜?            recent = conn.execute("""
                SELECT evidence_contributions FROM analysis_records
                ORDER BY created_at DESC LIMIT 10
            """).fetchall()

            source_aggregate = {}
            for row in recent:
                import json
                contributions = json.loads(row[0]) if row[0] else {}
                for source, info in contributions.items():
                    if source not in source_aggregate:
                        source_aggregate[source] = {"total_pct": 0, "count": 0}
                    source_aggregate[source]["total_pct"] += info.get("percentage", 0)
                    source_aggregate[source]["count"] += 1

            return {
                source: {
                    "avg_percentage": round(info["total_pct"] / info["count"], 1),
                    "appearances": info["count"],
                }
                for source, info in sorted(
                    source_aggregate.items(),
                    key=lambda x: -x[1]["total_pct"] / x[1]["count"]
                )
            }
        finally:
            conn.close()

    def suggest_weight_adjustment(self) -> dict:
        """
        鍩轰簬鍘嗗彶鏁版嵁寤鸿鏉冮噸璋冩暣銆?        褰撴煇璇佹嵁婧愮殑璐＄尞鐜囨寔缁綆浜庡叾鏉冮噸鏃讹紝鍙兘闇€瑕侀檷浣庡叾鏉冮噸銆?

        overview = self.get_evidence_source_overview()
        suggestions = []

        # 鏉冮噸鍙傝€冿紙涓?MergeWorker 涓殑 SOURCE_WEIGHTS 涓€鑷达級
        reference_weights = {
            "sql_join": 0.40,
            "orm_association": 0.25,
            "orm_collection": 0.25,
            "column_name_suffix": 0.20,
            "primary_key_name_match": 0.20,
            "naming_convention_match": 0.18,
            "naming_cross_table": 0.15,
        }

        for source, info in overview.items():
            weight = reference_weights.get(source, 0.10)
            if info["avg_percentage"] < weight * 30:  # 璐＄尞鐜囪繙浣庝簬鏉冮噸
                suggestions.append({
                    "source": source,
                    "current_weight": weight,
                    "avg_contribution": info["avg_percentage"],
                    "suggestion": f"璐＄尞鐜?{info['avg_percentage']}%)杩滀綆浜庢潈閲?{weight*100}%)锛屽缓璁檷浣庢潈閲?,
                })

        return {
            "evidence_source_overview": overview,
            "adjustment_suggestions": suggestions,
            "note": "姝や负寤鸿锛屽疄闄呮潈閲嶈皟鏁撮渶浜哄伐纭鍚庝慨鏀?MergeWorker 鐨?SOURCE_WEIGHTS",
        }

                "source_table": rel.get("source_table", "").lower(),
                "target_table": rel.get("target_table", "").lower(),
                "fk_column": rel.get("fk_column", "").lower(),
                "pk_column": rel.get("pk_column", "").lower(),

        for exp in expected:

                "fk_column": exp["fk_column"].lower(),

            key = (det["source_table"], det["fk_column"])
            if key not in expected_keys and det["confidence"] >= 0.7:
                high_confidence_wrong += 1

        # 5. 璁＄畻缁撴灉
        total_expected = len(expected_normalized)
        total_detected = len(detected_relations)
        total_high = len(high)

        precision = high_confidence_correct / (high_confidence_correct + high_confidence_wrong) \
            if (high_confidence_correct + high_confidence_wrong) > 0 else 0
        recall = high_confidence_correct / total_expected if total_expected > 0 else 0
        f1 = 2 * precision * recall / (precision + recall) if (precision + recall) > 0 else 0

        # 6. 鐢熸垚鎶ュ憡
        report = {
            "test_info": {
                "total_expected_relations": total_expected,
                "total_detected_relations": total_detected,
            },
            "scores": {

                "recall": round(recall, 4),
                "f1_score": round(f1, 4),
            },

                "high_confidence_wrong": high_confidence_wrong,
                "medium_confidence_correct": medium_confidence_correct,

            },

                "test_set_version": test_data.get("meta", {}).get("version", "1.0"),
            }
        }
