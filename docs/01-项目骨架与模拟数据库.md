import pymysql

from backend.config import Config


class MySQLSimulator:
    """鏁版嵁搴撹繛鎺ュ拰鏌ヨ灏佽"""

    _pool = None

    @classmethod
    def get_pool(cls):
        if cls._pool is None:
            cls._pool = PooledDB(
                creator=pymysql,
                host=Config.DB_HOST,
                port=Config.DB_PORT,
                user=Config.DB_USER,
                password=Config.DB_PASSWORD,
                database=Config.DB_NAME,
                charset='utf8mb4',
                maxconnections=10,
                mincached=2,
                blocking=True,
            )
        return cls._pool

    @classmethod
    def execute_query(cls, sql: str, params=None) -> list[dict]:
        """鎵ц SELECT 鏌ヨ锛岃繑鍥?list[dict]"""
        pool = cls.get_pool()
        conn = pool.connection()
        try:
            with conn.cursor(pymysql.cursors.DictCursor) as cursor:
                cursor.execute(sql, params or ())
                return cursor.fetchall()
        finally:
            conn.close()

    @classmethod
    def execute_dict(cls, sql: str, params=None) -> list[dict]:
        """鎵ц SQL 骞惰繑鍥炲瓧鍏稿垪琛紙鍚?execute_query 鐨勫埆鍚嶏級"""
